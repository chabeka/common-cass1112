Plusieurs commandes sont disponibles :
- listeOrgs 		:	lister les organismes présents dans l'ensemble des documents
- listeDocs 		: 	lister les documents dont au moins un des codes organisme est 
			  			présent dans le fichier de correspondance.
- maj       		:	mise à jour des documents avec les correspondances des codes
						organismes
- listeNonIntegres	: 	lister les documents dont au moins un des codes organisme est 
			  			présent dans le fichier de correspondance.

				
// listeOrgs
commande :
java -jar -Dlogback.configurationFile=[logback] sae-regionalisation-fond-documentaire.jar listeOrgs [connexions] [sortie]
avec :
[logback] 		= fichier de configuration des logs logback
[connexions] 	= fichier de configuration des connexions DFCE + CASSANDRA
[sortie]		= fichier où sera inscrit les données


// listeDocs
commande :
java -jar -Dlogback.configurationFile=[logback] sae-regionalisation-fond-documentaire.jar listeDocs [connexions] [sortie] [corresp]
avec :
[logback] 		= fichier de configuration des logs logback
[connexions] 	= fichier de configuration des connexions DFCE + CASSANDRA
[sortie]		= fichier où sera inscrit les données
[corresp]		= fichier properties contenant la liste des correspondances de codes organismes

// maj
commande :
java -jar -Dlogback.configurationFile=[logback] sae-regionalisation-fond-documentaire.jar maj [connexions] [donnees] [corresp] [first_index] [last_index]
avec :
[logback] 		= fichier de configuration des logs logback
[connexions] 	= fichier de configuration des connexions DFCE + CASSANDRA
[donnees}		= fichier de données à traiter
[corresp]		= fichier properties contenant la liste des correspondances de codes organismes
[first_index]	= index du premier enregistrement à traiter
[last_index]	= index du dernier enregistrement à traiter

// listeNonIntegres
commande :
java -jar -Dlogback.configurationFile=[logback] sae-regionalisation-fond-documentaire.jar listeNonIntegres [connexions] [sortie] [corresp]
avec :
[logback] 		= fichier de configuration des logs logback
[connexions] 	= fichier de configuration des connexions DFCE + CASSANDRA
[sortie]		= fichier où sera inscrit les données
[corresp]		= fichier properties contenant la liste des correspondances de codes organismes
